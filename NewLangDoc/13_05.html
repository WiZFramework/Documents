#pragma once

namespace clg
{
    //--------------------------------------------------------------------------------------
    ///  ニーモニック
    //--------------------------------------------------------------------------------------
    enum class Mnemonic {
        LDI,    //INT型定数のロード
        STO,    //識別子のセット
        DMP,    //デバッグ出力
        MUL,
        DIV,
        ADD,
        SUB,
        ASS,
        DCI     //int宣言
    };

    //--------------------------------------------------------------------------------------
    ///  値タイプ
    //--------------------------------------------------------------------------------------
    enum class ValueType {
        intType,
        indexType,
    };

    //--------------------------------------------------------------------------------------
    ///  値
    //--------------------------------------------------------------------------------------
    struct Value {
        ValueType valType;
        union {
            int intVal;
            char* pChar;
        };
    };

    //--------------------------------------------------------------------------------------
    ///  オペコード
    //--------------------------------------------------------------------------------------
    struct Opecode {
        Mnemonic mmnic;
        Value val;
    };


    //--------------------------------------------------------------------------------------
    ///  ランタイムスコープ
    //--------------------------------------------------------------------------------------
    struct RuntimeScope {
        map<string, Value> m_variableMap;
    };


    //--------------------------------------------------------------------------------------
    ///  スタックマシンクラス
    //--------------------------------------------------------------------------------------
    class stackMachine {
        //文字列テーブル
        vector<string> m_strings;
        vector<Opecode> m_codes;
        vector<Value> m_stack;
        vector<shared_ptr<RuntimeScope>> m_scopeStack;
        //命令ポインタ
        unsigned int m_ip = 0;
        void stackPush(Value val);
        Value stackPop();
        shared_ptr<RuntimeScope> tgtScope() const;
        void scopePush();
        void scopePop();
        void binaryExec(Mnemonic nmc);
        int getValInt(const Value& val) const;
        //文字列テーブルを参照して存在すればそのインデックス、
        //無ければ作成してインデックスを返す
        int chkStringTable(const char* ptr);
        //変数の存在を確認して見つかればそのスコープ
        //見つからなければnullptrを返す
        shared_ptr<RuntimeScope> findVariableScope(const Value& val)const;
        stackMachine();
    public:
        virtual ~stackMachine();
        int compile(FILE* fp);
        void exec();
        void destroy();
        //インスタンス参照
        static stackMachine* get();
        //何もしない
        void createEmpty();
        /// INTリテラル
        void createIntLiteralExp(const char* ptr);
        //識別子
        void createIdentifierExp(const char* ptr);
        //代入
        void createAssExp();
        //四則演算
        void createBinaryExp(Mnemonic nmc);
        //int型宣言
        void createIntVariableDecl();
        //デバッグ出力
        void createDump();
    };
}
//end namespace clg

stackMachine.cpp

#include "proc.h"

namespace clg
{
    //唯一のスタックマシンのインスタンス
    stackMachine* g_pSackMachine;

    stackMachine::stackMachine()
    {
        auto ptr = shared_ptr<RuntimeScope>(new RuntimeScope);
        m_scopeStack.push_back(ptr);
    }
    stackMachine::~stackMachine() {
    }
    int stackMachine::compile(FILE* fp) {
        extern int yyparse(void);
        extern FILE* yyin;
        yyin = fp;
        if (yyparse()) {
            return 1;
        }
        return 0;

    }

    void stackMachine::binaryExec(Mnemonic nmc) {
        Value val;
        val = stackPop();
        if (m_stack.size() > 0) {
            switch (nmc) {
            case Mnemonic::MUL:
                m_stack[m_stack.size() - 1].intVal *= val.intVal;
                break;
            case Mnemonic::DIV:
                m_stack[m_stack.size() - 1].intVal /= val.intVal;
                break;
            case Mnemonic::ADD:
                m_stack[m_stack.size() - 1].intVal += val.intVal;
                break;
            case Mnemonic::SUB:
                m_stack[m_stack.size() - 1].intVal -= val.intVal;
                break;
            }
        }
    }

    //文字列テーブルを参照して存在すればそのインデックス、
    //無ければ作成してインデックスを返す
    int stackMachine::chkStringTable(const char* ptr) {
        string str = ptr;
        int ret;
        auto it = std::find(m_strings.begin(), m_strings.end(), str);
        if (it == m_strings.end()) {
            ret = m_strings.size();
            m_strings.push_back(str);
        }
        else {
            ret = std::distance(m_strings.begin(), it);
        }
        return ret;
    }

    int stackMachine::getValInt(const Value& val) const {
        shared_ptr<RuntimeScope> scope;
        switch (val.valType) {
        case ValueType::intType:
            return val.intVal;
            break;
        case ValueType::indexType:
            scope = findVariableScope(val);
            if (!scope) {
                cout << m_strings[val.intVal] << ": 変数が宣言されていません" << endl;
                exit(1);
            }
            return scope->m_variableMap[m_strings[val.intVal]].intVal;
            break;
        }
        return 0;
    }

    void stackMachine::exec() {
        if (m_codes.size() <= 0) {
            return;
        }
        while (m_ip < m_codes.size()) {
            Opecode opcode = m_codes[m_ip];
            switch (opcode.mmnic) {
            case Mnemonic::LDI:
            case Mnemonic::STO:
                stackPush(opcode.val);
                m_ip++;
                break;
            case Mnemonic::MUL:
            case Mnemonic::DIV:
            case Mnemonic::ADD:
            case Mnemonic::SUB:
                binaryExec(opcode.mmnic);
                m_ip++;
                break;
            case Mnemonic::DCI:
            {
                auto val = stackPop();
                auto scope = tgtScope();
                //とりあえず0初期化
                Value val2;
                val2.valType = ValueType::intType;
                val2.intVal = 0;
                scope->m_variableMap[m_strings[val.intVal]] = val2;
                m_ip++;
            }
                break;
            case Mnemonic::ASS:
            {
                auto val1 = stackPop();
                auto val2 = stackPop();
                int i = getValInt(val1);
                val1.valType = ValueType::intType;
                val1.intVal = i;
                auto tgtscope = findVariableScope(val2);
                if (!tgtscope) {
                    cout << m_strings[val2.intVal] << ": 変数が宣言されていません" << endl;
                    exit(1);
                }
                tgtscope->m_variableMap[m_strings[val2.intVal]] = val1;
                //代入演算子は戻り値は代入した値そのものなので、代入した値をスタックに戻しておく。
                stackPush(val1);
            }
                m_ip++;
                break;
            case Mnemonic::DMP:
            {
                cout << getValInt(stackPop()) << endl;
            }
                m_ip++;
                break;
            }
        }
    }

    void stackMachine::destroy() {
        if (g_pSackMachine) {
            delete g_pSackMachine;
            g_pSackMachine = nullptr;
        }
    }

    //インスタンス参照
    stackMachine* stackMachine::get() {
        if (!g_pSackMachine) {
            g_pSackMachine = new stackMachine();
        }
        return g_pSackMachine;
    }
    //スタック操作
    void stackMachine::stackPush(Value val) {
        m_stack.push_back(val);
    }
    Value stackMachine::stackPop() {
        if (m_stack.empty()) {
            cout << "スタックは空です" << endl;
            exit(1);
        }
        Value val = m_stack.back();
        m_stack.pop_back();
        return val;
    }
    void stackMachine::scopePush() {
        auto ptr = shared_ptr<RuntimeScope>(new RuntimeScope);
        m_scopeStack.push_back(ptr);
    }
    void stackMachine::scopePop() {
        if (m_scopeStack.size() <= 0) {
            cout << "スコープスタックは空です" << endl;
            exit(1);
        }
        m_scopeStack.pop_back();
    }

    shared_ptr<RuntimeScope> stackMachine::tgtScope()const {
        if (m_scopeStack.size() <= 0) {
            cout << "スコープスタックは空です" << endl;
            exit(1);
        }
        return m_scopeStack[m_scopeStack.size() - 1];
    }

    shared_ptr<RuntimeScope> stackMachine::findVariableScope(const Value& val)const {
        if (m_scopeStack.size() <= 0) {
            cout << "スコープスタックは空です" << endl;
            exit(1);
        }
        int i = m_scopeStack.size() - 1;
        while (i >= 0) {
            if (m_strings.size() <= (unsigned int)val.intVal) {
                cout << "文字列テーブルに該当の識別子はありません" << endl;
                exit(1);

            }
            string str = m_strings[val.intVal];
            auto it = m_scopeStack[i]->m_variableMap.find(str);
            if (it != m_scopeStack[i]->m_variableMap.end()) {
                //見つかった
                return m_scopeStack[i];
            }
            i--;
        }
        //見つからなかった
        return nullptr;
    }



    //何もしない
    void stackMachine::createEmpty() {

    }

    //四則演算
    void stackMachine::createBinaryExp(Mnemonic nmc) {
        Opecode code;
        code.mmnic = nmc;
        m_codes.push_back(code);
    }

    /// INTリテラル
    void stackMachine::createIntLiteralExp(const char* ptr) {
        Opecode code;
        code.mmnic = Mnemonic::LDI;
        code.val.valType = ValueType::intType;
        code.val.intVal = atoi(ptr);
        m_codes.push_back(code);
    }

    //識別子
    void stackMachine::createIdentifierExp(const char* ptr) {
        Opecode code;
        code.mmnic = Mnemonic::STO;
        //ストリングテーブルに保存またはインデックスの取得
        //実際の変数操作は代入が行われたとき作成
        code.val.valType = ValueType::indexType;
        code.val.intVal = chkStringTable(ptr);
        m_codes.push_back(code);
    }

    //代入
    void stackMachine::createAssExp() {
        Opecode code;
        code.mmnic = Mnemonic::ASS;
        m_codes.push_back(code);
    }
    //宣言
    void stackMachine::createIntVariableDecl() {
        Opecode code;
        code.mmnic = Mnemonic::DCI;
        m_codes.push_back(code);
    }
    //出力
    void stackMachine::createDump() {
        Opecode code;
        code.mmnic = Mnemonic::DMP;
        m_codes.push_back(code);
    }
}

main.cpp

#include "proc.h"

class InputParser {
    vector <string> m_tokens;
public:
    InputParser(int& argc, char** argv) {
        for (int i = 1; i < argc; ++i) {
            m_tokens.push_back(string(argv[i]));
        }
    }
    const string& getCmdOption(const string& option) const {
        vector<string>::const_iterator itr;
        itr = find(m_tokens.begin(), m_tokens.end(), option);
        if (itr != m_tokens.end() && ++itr != m_tokens.end()) {
            return *itr;
        }
        static const string empty_string("");
        return empty_string;
    }
    bool cmdOptionExists(const string& option) const {
        return find(m_tokens.begin(), m_tokens.end(), option)
            != m_tokens.end();
    }
};


int main(int argc, char** argv) {
    InputParser input(argc, argv);
    const string& filename = input.getCmdOption("-f");
    if (filename.empty()) {
        cout << "スクリプトファイルが指定されていません。" << endl;
        return 1;
    }
    errno_t err;
    FILE* fp;
    if ((err = fopen_s(&fp, filename.c_str(), "r")) != 0) {
        cout << "スクリプトファイルの読み込みに失敗しました。" << endl;
        return 1;
    }
    auto tntp = clg::stackMachine::get();
    if (tntp->compile(fp)) {
        return 1;
    }
    tntp->exec();
    tntp->destroy();
    return 0;
}
