653  void DrawSprite(const MeshPrimData& data) {
654  auto Dev = App::GetApp()->GetDeviceResources();
655  auto pD3D11DeviceContext = Dev->GetD3DDeviceContext();
656  auto RenderState = Dev->GetRenderState();
657  //NULLのシェーダリソースの準備
658  ID3D11ShaderResourceView* pNull[D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT] = { nullptr };
659  //サンプラーの準備
660  ID3D11SamplerState* pNullSR[D3D11_COMMONSHADER_SAMPLER_SLOT_COUNT] = { nullptr };
661  //各オブジェクト共通処理
662  //シェーダの設定
663  //頂点シェーダ
664  pD3D11DeviceContext->VSSetShader(T_VShader::GetPtr()->GetShader(), nullptr, 0);
665  //インプットレイアウトの設定
666  pD3D11DeviceContext->IASetInputLayout(T_VShader::GetPtr()->GetInputLayout());
667  //ピクセルシェーダ
668  pD3D11DeviceContext->PSSetShader(T_PShader::GetPtr()->GetShader(), nullptr, 0);
669  //個別処理
670  SpriteConstants sb;
671  //コンスタントバッファの作成
672  SetConstants(sb);
673  //テクスチャ
674  auto shTex = GetTextureResource();
675  //コンスタントバッファの更新
676  pD3D11DeviceContext->UpdateSubresource(CBSprite::GetPtr()->GetBuffer(), 0, nullptr, &sb, 0, 0);
677  //コンスタントバッファの設定
678  ID3D11Buffer* pConstantBuffer = CBSprite::GetPtr()->GetBuffer();
679  ID3D11Buffer* pNullConstantBuffer = nullptr;
680  //頂点シェーダに渡す
681  pD3D11DeviceContext->VSSetConstantBuffers(0, 1, &pConstantBuffer);
682  //ピクセルシェーダに渡す
683  pD3D11DeviceContext->PSSetConstantBuffers(0, 1, &pConstantBuffer);
684  //ストライドとオフセット
685  UINT stride = data.m_NumStride;
686  UINT offset = 0;
687  //描画方法のセット
688  pD3D11DeviceContext->IASetPrimitiveTopology(data.m_PrimitiveTopology);
689  //頂点バッファのセット
690  pD3D11DeviceContext->IASetVertexBuffers(0, 1, data.m_VertexBuffer.GetAddressOf(), &stride, &offset);
691  //インデックスバッファのセット
692  pD3D11DeviceContext->IASetIndexBuffer(data.m_IndexBuffer.Get(), DXGI_FORMAT_R16_UINT, 0);
693  //各レンダリングステートの設定
694  //ブレンドステート
695  RenderState->SetBlendState(pD3D11DeviceContext, GetBlendState());
696  //デプスステンシルステート
697  RenderState->SetDepthStencilState(pD3D11DeviceContext, GetDepthStencilState());
698  //テクスチャとサンプラー
699  if (shTex) {
700  pD3D11DeviceContext->PSSetShaderResources(0, 1, shTex->GetShaderResourceView().GetAddressOf());
701  //サンプラーを設定
702  RenderState->SetSamplerState(pD3D11DeviceContext, GetSamplerState(), 0);
703  }
704  else {
705  //シェーダーリソースもクリア
706  pD3D11DeviceContext->PSSetShaderResources(0, D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT, pNull);
707  //サンプラーもクリア
708  RenderState->SetSamplerAllClear(pD3D11DeviceContext);
709  }
710  //ラスタライザステートと描画
711  RenderState->SetRasterizerState(pD3D11DeviceContext, GetRasterizerState());
712  //描画
713  pD3D11DeviceContext->DrawIndexed(data.m_NumIndicis, 0, 0);
714  }
