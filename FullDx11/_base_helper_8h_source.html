1358  m_qpcSecondCounter += timeDelta;
1359

1361  if (timeDelta > m_qpcMaxDelta)

1367  timeDelta *= TicksPerSecond;
1368  timeDelta /= m_qpcFrequency.QuadPart;
1369 
1370  uint32 lastFrameCount = m_frameCount;
1371 
1372  if (m_isFixedTimeStep)
1373  {
1374  // 固定タイムステップ更新ロジック
1375 
1376  // クランプしたターゲット経過時間 (1/4 ミリ秒以内) に接近して、アプリケーションが実行されている場合
1377  // ターゲット値に正確に一致させるクロック。これにより、小さい無関係なエラーの発生を防ぎます。
1378  // 一定期間内に蓄積された。このクランプなしで、60 fps を要求したゲームは
1379  // 固定された更新 (59.94 NTSC ディスプレイ上で有効な vsync で実行) は最終的には、
1380  // フレームをドロップするのに十分な小さいエラーを蓄積します。丸めることをお勧めします
1381  // わずかな偏差をゼロに持っていき、スムーズに進行するようにします。
1382

1384  {
1385  timeDelta = m_targetElapsedTicks;
1386  }
1387 
1388  m_leftOverTicks += timeDelta;
1389 
1390  while (m_leftOverTicks >= m_targetElapsedTicks)
1391  {
1392  m_elapsedTicks = m_targetElapsedTicks;
1393  m_totalTicks += m_targetElapsedTicks;
1394  m_leftOverTicks -= m_targetElapsedTicks;
1395  m_frameCount++;
1396 
1397  update();
1398  }
1399  }
1400  else
1401  {
1402  // 可変タイムステップ更新ロジック。
1403  m_elapsedTicks = timeDelta;
1404  m_totalTicks += timeDelta;
1405  m_leftOverTicks = 0;
1406  m_frameCount++;
1407

1409  }

1415  }

1421  m_qpcSecondCounter %= m_qpcFrequency.QuadPart;
1422  }
