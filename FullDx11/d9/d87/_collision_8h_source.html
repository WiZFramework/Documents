371  template<typename SrcType, typename DestType>
372  void AfterCollisionTmpl_1(const bsm::Vec3& SrcVelocity, const bsm::Vec3& DestVelocity, float ElapsedTime, float HitTime,
373  const shared_ptr<SrcType>& SrcColl, const shared_ptr<DestType>& DestColl) {
374  if (HitTime <= 0) {
375  HitTime = 0;
376  }
377  if (HitTime >= ElapsedTime) {
378  HitTime = ElapsedTime;
379  }
380  auto AfterHitTime = ElapsedTime - HitTime;
381  bool UseSrcAfter = false;
382  bool UseDestAfter = false;
383  //衝突相手の登録
384  AddHitObject(DestColl->GetGameObject());
385  if (GetIsHitAction() != IsHitAction::None) {
386  //事後処理が有効なら衝突前まで戻る
387  BackToBefore(SrcVelocity, HitTime);
388  UseSrcAfter = true;
389  }
390  if (!DestColl->IsFixed()) {
391  //もし相手がFixでなければ相手の衝突相手の登録
392  DestColl->AddHitObject(GetGameObject());
393  if (DestColl->GetIsHitAction() != IsHitAction::None) {
394  //事後処理が有効なら衝突前まで戻る
395  DestColl->BackToBefore(DestVelocity, HitTime);
396  UseDestAfter = true;
397  }
398  }
399  else {
400  //相手がFixなら自己退避だけでは衝突している可能性があるので判定をしてエスケープ処理
401  if (GetIsHitAction() != IsHitAction::None) {
402  CollisionEscape(DestColl);
403  }
404  }
405  if (UseSrcAfter) {
406  bsm::Vec3 HitNormal;
407  GetHitNormal(DestColl, HitNormal);
408  AfterCollision(DestColl, SrcVelocity, DestVelocity, HitNormal, AfterHitTime);
409  }
410  if (UseDestAfter) {
411  bsm::Vec3 HitNormal;
412  DestColl->GetHitNormal(SrcColl, HitNormal);
413  DestColl->AfterCollision(SrcColl, DestVelocity, SrcVelocity, HitNormal, AfterHitTime);
414  }
415 
416  }
