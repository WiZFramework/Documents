199  template<typename T_VShader, typename T_PShader>
200  void DrawStatic(const MeshPrimData& data) {
201  auto Dev = App::GetApp()->GetDeviceResources();
202  auto pD3D11DeviceContext = Dev->GetD3DDeviceContext();
203  auto RenderState = Dev->GetRenderState();
204  //NULLのシェーダリソースの準備
205  ID3D11ShaderResourceView* pNull[D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT] = { nullptr };
206  //サンプラーの準備
207  ID3D11SamplerState* pNullSR[D3D11_COMMONSHADER_SAMPLER_SLOT_COUNT] = { nullptr };
208  //各オブジェクト共通処理
209  //シェーダの設定
210  //頂点シェーダ
211  pD3D11DeviceContext->VSSetShader(T_VShader::GetPtr()->GetShader(), nullptr, 0);
212  //インプットレイアウトの設定
213  pD3D11DeviceContext->IASetInputLayout(T_VShader::GetPtr()->GetInputLayout());
214  //ピクセルシェーダ
215  pD3D11DeviceContext->PSSetShader(T_PShader::GetPtr()->GetShader(), nullptr, 0);
216  //ライトを設定
217  SetLightingParamaters();
218  //個別処理
219  BasicConstants BcCb;
220  //コンスタントバッファの作成
221  SetConstants(BcCb, data);
222  //テクスチャ
223  auto shTex = GetTextureResource();
224  if (shTex) {
225  //テクスチャがある
226  BcCb.activeFlg.y = 1;
227  }
228  else {
229  //描画コンポーネントにはテクスチャがない
230  if (shTex = data.m_TextureResource.lock()) {
231  //テクスチャがある
232  BcCb.activeFlg.y = 1;
233  }
234  else {
235  BcCb.activeFlg.y = 0;
236  }
237  }
238  //コンスタントバッファの更新
239  pD3D11DeviceContext->UpdateSubresource(CBBasic::GetPtr()->GetBuffer(), 0, nullptr, &BcCb, 0, 0);
240  //コンスタントバッファの設定
241  ID3D11Buffer* pConstantBuffer = CBBasic::GetPtr()->GetBuffer();
242  ID3D11Buffer* pNullConstantBuffer = nullptr;
243  //頂点シェーダに渡す
244  pD3D11DeviceContext->VSSetConstantBuffers(0, 1, &pConstantBuffer);
245  //ピクセルシェーダに渡す
246  pD3D11DeviceContext->PSSetConstantBuffers(0, 1, &pConstantBuffer);
247  //ストライドとオフセット
248  UINT stride = data.m_NumStride;
249  UINT offset = 0;
250  //描画方法のセット
251  pD3D11DeviceContext->IASetPrimitiveTopology(data.m_PrimitiveTopology);
252  //頂点バッファのセット
253  pD3D11DeviceContext->IASetVertexBuffers(0, 1, data.m_VertexBuffer.GetAddressOf(), &stride, &offset);
254  //インデックスバッファのセット
255  pD3D11DeviceContext->IASetIndexBuffer(data.m_IndexBuffer.Get(), DXGI_FORMAT_R16_UINT, 0);
256  //各レンダリングステートの設定
257  //ブレンドステート
258  RenderState->SetBlendState(pD3D11DeviceContext, GetBlendState());
259  //デプスステンシルステート
260  RenderState->SetDepthStencilState(pD3D11DeviceContext, GetDepthStencilState());
261  //テクスチャとサンプラー
262  if (shTex) {
263  pD3D11DeviceContext->PSSetShaderResources(0, 1, shTex->GetShaderResourceView().GetAddressOf());
264  //サンプラーを設定
265  RenderState->SetSamplerState(pD3D11DeviceContext, GetSamplerState(), 0);
266  //法線マップ
267  if (GetNormalMapTextureResource()) {
268  if (IsOwnShadowActive()) {
269  //２番目に入れる
270  pD3D11DeviceContext->PSSetShaderResources(2, 1, GetNormalMapTextureResource()->GetShaderResourceView().GetAddressOf());
271  //サンプラーを設定
272  RenderState->SetSamplerState(pD3D11DeviceContext, GetSamplerState(), 2);
273  }
274  else {
275  //1番目に入れる
276  pD3D11DeviceContext->PSSetShaderResources(1, 1, GetNormalMapTextureResource()->GetShaderResourceView().GetAddressOf());
277  //サンプラーを設定
278  RenderState->SetSamplerState(pD3D11DeviceContext, GetSamplerState(), 1);
279  }
280  }
281  }
282  else {
283  //シェーダーリソースもクリア
284  pD3D11DeviceContext->PSSetShaderResources(0, D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT, pNull);
285  //サンプラーもクリア
286  RenderState->SetSamplerAllClear(pD3D11DeviceContext);
287  }
288  //影とサンプラー
289  if (IsOwnShadowActive()) {
290  //シャドウマップのレンダラーターゲット
291  auto ShadowmapPtr = Dev->GetShadowMapRenderTarget();
292  ID3D11ShaderResourceView* pShadowSRV = ShadowmapPtr->GetShaderResourceView();
293  pD3D11DeviceContext->PSSetShaderResources(1, 1, &pShadowSRV);
294  //シャドウマップサンプラー
295  ID3D11SamplerState* pShadowSampler = RenderState->GetComparisonLinear();
296  pD3D11DeviceContext->PSSetSamplers(1, 1, &pShadowSampler);
297  }
298  //ラスタライザステートと描画
299  if (GetRasterizerState() == RasterizerState::DoubleDraw) {
300  //透明処理用
301  //ラスタライザステート(裏描画)
302  pD3D11DeviceContext->RSSetState(RenderState->GetCullFront());
303  //描画
304  pD3D11DeviceContext->DrawIndexed(data.m_NumIndicis, 0, 0);
305  //ラスタライザステート（表描画）
306  pD3D11DeviceContext->RSSetState(RenderState->GetCullBack());
307  //描画
308  pD3D11DeviceContext->DrawIndexed(data.m_NumIndicis, 0, 0);
309  }
310  else {
311  RenderState->SetRasterizerState(pD3D11DeviceContext, GetRasterizerState());
312  //描画
313  pD3D11DeviceContext->DrawIndexed(data.m_NumIndicis, 0, 0);
314  }
315  //後始末
316 // Dev->InitializeStates();
317  }

327  template<typename T_VShader, typename T_PShader>
328  void DrawModel(const MeshPrimData& data) {
329  auto Dev = App::GetApp()->GetDeviceResources();
330  auto pD3D11DeviceContext = Dev->GetD3DDeviceContext();
331  auto RenderState = Dev->GetRenderState();
332  //NULLのシェーダリソースの準備
333  ID3D11ShaderResourceView* pNull[D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT] = { nullptr };
334  //サンプラーの準備
335  ID3D11SamplerState* pSamplerClamp = RenderState->GetLinearClamp();
336  ID3D11SamplerState* pSamplerWrap = RenderState->GetLinearWrap();
337  ID3D11SamplerState* pNullSR[D3D11_COMMONSHADER_SAMPLER_SLOT_COUNT] = { nullptr };
338  //各オブジェクト共通処理
339  //シェーダの設定
340  //頂点シェーダ
341  pD3D11DeviceContext->VSSetShader(T_VShader::GetPtr()->GetShader(), nullptr, 0);
342  //インプットレイアウトの設定
343  pD3D11DeviceContext->IASetInputLayout(T_VShader::GetPtr()->GetInputLayout());
344  //ピクセルシェーダ
345  pD3D11DeviceContext->PSSetShader(T_PShader::GetPtr()->GetShader(), nullptr, 0);
346  //ストライドとオフセット
347  UINT stride = data.m_NumStride;
348  UINT offset = 0;
349  //描画方法のセット
350  pD3D11DeviceContext->IASetPrimitiveTopology(data.m_PrimitiveTopology);
351  //頂点バッファのセット
352  pD3D11DeviceContext->IASetVertexBuffers(0, 1, data.m_VertexBuffer.GetAddressOf(), &stride, &offset);
353  //インデックスバッファのセット
354  pD3D11DeviceContext->IASetIndexBuffer(data.m_IndexBuffer.Get(), DXGI_FORMAT_R16_UINT, 0);
355  //各レンダリングステートの設定
356  //ブレンドステート
357  RenderState->SetBlendState(pD3D11DeviceContext, GetBlendState());
358  //デプスステンシルステート
359  RenderState->SetDepthStencilState(pD3D11DeviceContext, GetDepthStencilState());
360  //影とサンプラー
361  if (GetOwnShadowActive()) {
362  //シャドウマップのレンダラーターゲット
363  auto ShadowmapPtr = Dev->GetShadowMapRenderTarget();
364  ID3D11ShaderResourceView* pShadowSRV = ShadowmapPtr->GetShaderResourceView();
365  pD3D11DeviceContext->PSSetShaderResources(1, 1, &pShadowSRV);
366  //シャドウマップサンプラー
367  ID3D11SamplerState* pShadowSampler = RenderState->GetComparisonLinear();
368  pD3D11DeviceContext->PSSetSamplers(1, 1, &pShadowSampler);
369  }
370  //ライトを設定
371  SetLightingParamaters();
372  BasicConstants BcCb;
373  //コンスタントバッファの作成
374  SetConstants(BcCb, data);
375  //メッシュ内のマテリアルの配列
376  auto& MatVec = data.m_MaterialExVec;
377  size_t MatIndex = 0;
378  for (auto& m : MatVec) {
379  if (m.m_TextureResource) {
380  //テクスチャがある
381  BcCb.activeFlg.y = 1;
382  }
383  else {
384  //テクスチャがない
385  BcCb.activeFlg.y = 0;
386  }
387  if (IsModelDiffusePriority()) {
388  BcCb.diffuseColor = m.m_Diffuse;
389  }
390  if (IsModelEmissivePriority()) {
391  bsm::Col4 Em4 = m.m_Emissive;
392  Em4.w = 0.0f;
393  BcCb.emissiveColor = Em4;
394  }
395  //コンスタントバッファの更新
396  pD3D11DeviceContext->UpdateSubresource(CBBasic::GetPtr()->GetBuffer(), 0, nullptr, &BcCb, 0, 0);
397  //コンスタントバッファの設定
398  ID3D11Buffer* pConstantBuffer = CBBasic::GetPtr()->GetBuffer();
399  ID3D11Buffer* pNullConstantBuffer = nullptr;
400  //頂点シェーダに渡す
401  pD3D11DeviceContext->VSSetConstantBuffers(0, 1, &pConstantBuffer);
402  //ピクセルシェーダに渡す
403  pD3D11DeviceContext->PSSetConstantBuffers(0, 1, &pConstantBuffer);
404  if (BcCb.activeFlg.y) {
405  //シェーダにテクスチャの設定がされている
406  //サンプラーの設定
407  RenderState->SetSamplerState(pD3D11DeviceContext, GetSamplerState(), 0);
408  //テクスチャの選択
409  if (IsModelTextureEnabled()) {
410  //モデルのテクスチャが有効
411  pD3D11DeviceContext->PSSetShaderResources(0, 1, m.m_TextureResource->GetShaderResourceView().GetAddressOf());
412  }
413  else {
414  //モデルのテクスチャが無効
415  auto shTex = GetTextureResource();
416  if (shTex) {
417  //コンポーネントにテクスチャがある
418  pD3D11DeviceContext->PSSetShaderResources(0, 1, shTex->GetShaderResourceView().GetAddressOf());
419  }
420  else {
421  //コンポーネントにテクスチャがない
422  BcCb.activeFlg.y = 0;
423  }
424  }
425  if (BcCb.activeFlg.y) {
426  //法線マップ
427  if (GetNormalMapTextureResourceVec().size() > MatIndex) {
428  //複数の法線マップがある
429  auto& normalVec = GetNormalMapTextureResourceVec();
430  auto normalShptr = normalVec[MatIndex].lock();
431  if (GetOwnShadowActive()) {
432  //２番目に入れる
433  pD3D11DeviceContext->PSSetShaderResources(2, 1, normalShptr->GetShaderResourceView().GetAddressOf());
434  //サンプラーを設定
435  RenderState->SetSamplerState(pD3D11DeviceContext, GetSamplerState(), 2);
436  }
437  else {
438  //1番目に入れる
439  pD3D11DeviceContext->PSSetShaderResources(1, 1, normalShptr->GetShaderResourceView().GetAddressOf());
440  //サンプラーを設定
441  RenderState->SetSamplerState(pD3D11DeviceContext,GetSamplerState(), 1);
442  }
443  }
444  else if (GetNormalMapTextureResource()) {
445  //そうではなくて一つだけ法線マップがある
446  if (GetOwnShadowActive()) {
447  //２番目に入れる
448  pD3D11DeviceContext->PSSetShaderResources(2, 1, GetNormalMapTextureResource()->GetShaderResourceView().GetAddressOf());
449  //サンプラーを設定
450  RenderState->SetSamplerState(pD3D11DeviceContext, GetSamplerState(), 2);
451  }
452  else {
453  //1番目に入れる
454  pD3D11DeviceContext->PSSetShaderResources(1, 1, GetNormalMapTextureResource()->GetShaderResourceView().GetAddressOf());
455  //サンプラーを設定
456  RenderState->SetSamplerState(pD3D11DeviceContext, GetSamplerState(), 1);
457  }
458  }
459  }
460  else {
461  //シェーダーリソースもクリア
462  pD3D11DeviceContext->PSSetShaderResources(0, D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT, pNull);
463  //サンプラーもクリア
464  RenderState->SetSamplerAllClear(pD3D11DeviceContext);
465  }
466  }
467  else {
468  //シェーダーリソースもクリア
469  pD3D11DeviceContext->PSSetShaderResources(0, D3D11_COMMONSHADER_INPUT_RESOURCE_SLOT_COUNT, pNull);
470  //サンプラーもクリア
471  RenderState->SetSamplerAllClear(pD3D11DeviceContext);
472  }
473  //ラスタライザステートと描画
474  if (GetRasterizerState() == RasterizerState::DoubleDraw) {
475  //透明処理用
476  //ラスタライザステート(裏描画)
477  pD3D11DeviceContext->RSSetState(RenderState->GetCullFront());
478  //描画
479  pD3D11DeviceContext->DrawIndexed(m.m_IndexCount, m.m_StartIndex, 0);
480  //ラスタライザステート（表描画）
481  pD3D11DeviceContext->RSSetState(RenderState->GetCullBack());
482  //描画
483  pD3D11DeviceContext->DrawIndexed(m.m_IndexCount, m.m_StartIndex, 0);
484  }
485  else {
486  RenderState->SetRasterizerState(pD3D11DeviceContext, GetRasterizerState());
487  //描画
488  pD3D11DeviceContext->DrawIndexed(m.m_IndexCount, m.m_StartIndex, 0);
489  }
490  MatIndex++;
491  }
