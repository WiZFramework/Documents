510  // 安全にD3Dリソースを作成するヘルパー関数.
511  template<typename T, typename TCreateFunc>
512  static T* DemandCreate(Microsoft::WRL::ComPtr<T>& comPtr, std::mutex& mutex, TCreateFunc createFunc)
513  {
514  T* result = comPtr.Get();
515  // Double-checked lock pattern.
516  MemoryBarrier();
517  if (!result)
518  {
519  std::lock_guard<std::mutex> lock(mutex);
520  result = comPtr.Get();
521  if (!result)
522  {
523  createFunc(&result);
524  MemoryBarrier();
525  comPtr.Attach(result);
526  }
527  }
528  return result;
529  }
