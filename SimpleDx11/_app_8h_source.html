363  bool IsMouseEnabled(HWND hWnd, vector<DWORD>& UseKeyVec) {
364  //マウスポイントは0,0に初期化
365  m_MouseClientPoint = { 0,0 };
366  //マウス利用にかかわらず、マウスポインタ座標は設定
367  POINT p;
368  ::GetCursorPos(&p);
369  if (::ScreenToClient(hWnd, &p)) {
370  //クライアント座標に変換できたときのみ、マウス座標を設定
371  m_MouseClientPoint = { p.x, p.y };
372  }
373  else {
374  return false;
375  }
376 
377  //UseKeyVecにマウスがあった場合
378  vector<DWORD> MouseTemp = { VK_LBUTTON, VK_RBUTTON, VK_MBUTTON };
379  bool MouseChk = false;
380  for (auto chk : MouseTemp) {
381  auto it = find(UseKeyVec.begin(), UseKeyVec.end(), chk);
382  if (it != UseKeyVec.end()) {
383  MouseChk = true;
384  break;
385  }
386  }
387  //マウス使用で座標がクライアント領域内ならtrue
388  if (MouseChk) {
389  RECT rc;
390  ::GetClientRect(hWnd, &rc);
391  if (::PtInRect(&rc, p)) {
392  return true;
393  }
394  }
395  return false;
396  }

405  bool GetKeyState(HWND hWnd, vector<DWORD>& UseKeyVec) {
406  m_KeyMessageActive = false;
407  //一つ前にコピー
408  CopyMemory(m_bLastKeyTbl, m_bPushKeyTbl, sizeof(m_bLastKeyTbl));
409  //キーボードの状態を初期化
410  ::ZeroMemory(&m_bPushKeyTbl, sizeof(m_bPushKeyTbl));
411  ::ZeroMemory(&m_bPressedKeyTbl, sizeof(m_bPressedKeyTbl));
412  ::ZeroMemory(&m_bUpKeyTbl, sizeof(m_bUpKeyTbl));
413  //マウスのチェック
414  bool MouseEnabled = IsMouseEnabled(hWnd, UseKeyVec);
415  //自分自身にフォーカスがない場合はfalse
416  if (::GetFocus() != hWnd) {
417  return false;
418  }
419  SHORT ret = GetAsyncKeyState((int)VK_ESCAPE);
420  if (ret & 0x8000) {
421  //Escキーは無条件にfalse
422  return false;
423  }
424  size_t sz = UseKeyVec.size();
425  for (size_t i = 0; i < sz; i++) {
426  ret = GetAsyncKeyState((int)UseKeyVec[i]);
427  if (UseKeyVec[i] == VK_LBUTTON || UseKeyVec[i] == VK_RBUTTON || UseKeyVec[i] == VK_MBUTTON) {
428  if (!MouseEnabled) {
429  //マウスが無効ならUsedに入っていても無視
430  continue;
431  }
432  }
433  if (ret & 0x8000) {
434  m_bPushKeyTbl[UseKeyVec[i]] = true;
435  m_KeyMessageActive = true; //メッセージをゲームが受け取る
436  if (!m_bLastKeyTbl[UseKeyVec[i]]) {
437  //前回押されてなくて今回押された
438  m_bPressedKeyTbl[UseKeyVec[i]] = true;
439  }
440  }
441  else {
442  //キーは押されてない
443  if (m_bLastKeyTbl[UseKeyVec[i]]) {
444  //前回押されていて今回押されてない
445  m_bUpKeyTbl[UseKeyVec[i]] = true;
446  m_KeyMessageActive = true; //メッセージをゲームが受け取る
447  }
448  }
449  }
450  return m_KeyMessageActive;
451  }

498  void ResetControlerState() {
499  DWORD dwResult;
500  for (DWORD i = 0; i < MAX_CONTROLLERS; i++)
501  {
502  //一つ前の状態を保存
503  m_State[i].wLastButtons = m_State[i].Gamepad.wButtons;
504  // XInputから現在のステータスを得る
505  XINPUT_STATE State;
506  dwResult = XInputGetState(i, &State);
507  if (dwResult == ERROR_SUCCESS) {
508  m_State[i].dwPacketNumber = State.dwPacketNumber;
509  m_State[i].Gamepad = State.Gamepad;
510  m_State[i].bConnected = true;
511  //ボタンの設定
512  //コピー
513  m_State[i].wButtons = m_State[i].Gamepad.wButtons;
514  //更新ボタン
515  m_State[i].wNowUpdateButtons = m_State[i].Gamepad.wButtons;
516  m_State[i].bLeftTrigger = m_State[i].Gamepad.bLeftTrigger;
517  m_State[i].bRightTrigger = m_State[i].Gamepad.bRightTrigger;
518  //前回より変化しているボタンのみフラグを立てる
519  //XOR演算により実装
520  m_State[i].wNowUpdateButtons ^= m_State[i].wLastButtons;
521  //押された瞬間をセット
522  m_State[i].wPressedButtons = m_State[i].wNowUpdateButtons & m_State[i].wButtons;
523  //離された瞬間をセット
524  m_State[i].wReleasedButtons = m_State[i].wNowUpdateButtons & (m_State[i].wButtons ^ 0xffff);
525  //レバーのデッドゾーンの設定
526  if ((m_State[i].Gamepad.sThumbLX < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE &&
527  m_State[i].Gamepad.sThumbLX > -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE) &&
528  (m_State[i].Gamepad.sThumbLY < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE &&
529  m_State[i].Gamepad.sThumbLY > -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE))
530  {
531  m_State[i].Gamepad.sThumbLX = 0;
532  m_State[i].Gamepad.sThumbLY = 0;
533  }
534  if (m_State[i].Gamepad.sThumbLX >= 0) {
535  m_State[i].fThumbLX = (float)m_State[i].Gamepad.sThumbLX / (float)SHRT_MAX;
536  }
537  else {
538  m_State[i].fThumbLX = (float)m_State[i].Gamepad.sThumbLX / (float)(SHRT_MAX + 1);
539  }
540  if (m_State[i].Gamepad.sThumbLY >= 0) {
541  m_State[i].fThumbLY = (float)m_State[i].Gamepad.sThumbLY / (float)SHRT_MAX;
542  }
543  else {
544  m_State[i].fThumbLY = (float)m_State[i].Gamepad.sThumbLY / (float)(SHRT_MAX + 1);
545  }
546 
547  if ((m_State[i].Gamepad.sThumbRX < XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE &&
548  m_State[i].Gamepad.sThumbRX > -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE) &&
549  (m_State[i].Gamepad.sThumbRY < XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE &&
550  m_State[i].Gamepad.sThumbRY > -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE))
551  {
552  m_State[i].Gamepad.sThumbRX = 0;
553  m_State[i].Gamepad.sThumbRY = 0;
554  }
555  if (m_State[i].Gamepad.sThumbRX >= 0) {
556  m_State[i].fThumbRX = (float)m_State[i].Gamepad.sThumbRX / (float)SHRT_MAX;
557  }
558  else {
559  m_State[i].fThumbRX = (float)m_State[i].Gamepad.sThumbRX / (float)(SHRT_MAX + 1);
560  }
561  if (m_State[i].Gamepad.sThumbRY >= 0) {
562  m_State[i].fThumbRY = (float)m_State[i].Gamepad.sThumbRY / (float)SHRT_MAX;
563  }
564  else {
565  m_State[i].fThumbRY = (float)m_State[i].Gamepad.sThumbRY / (float)(SHRT_MAX + 1);
566  }
567  }
568  else {
569  m_State[i].bConnected = false;
570  }
571  }
572  }
