321  bool IsMouseEnabled(HWND hWnd, vector<DWORD>& UseKeyVec) {
322  //マウスポイントは0,0に初期化
323  m_MouseClientPoint = { 0,0 };
324  //マウス利用にかかわらず、マウスポインタ座標は設定
325  POINT p;
326  ::GetCursorPos(&p);
327  if (::ScreenToClient(hWnd, &p)) {
328  //クライアント座標に変換できたときのみ、マウス座標を設定
329  m_MouseClientPoint = { p.x, p.y };
330  }
331  else {
332  return false;
333  }
334 
335  //UseKeyVecにマウスがあった場合
336  vector<DWORD> MouseTemp = { VK_LBUTTON, VK_RBUTTON, VK_MBUTTON };
337  bool MouseChk = false;
338  for (auto chk : MouseTemp) {
339  auto it = find(UseKeyVec.begin(), UseKeyVec.end(), chk);
340  if (it != UseKeyVec.end()) {
341  MouseChk = true;
342  break;
343  }
344  }
345  //マウス使用で座標がクライアント領域内ならtrue
346  if (MouseChk) {
347  RECT rc;
348  ::GetClientRect(hWnd, &rc);
349  if (::PtInRect(&rc, p)) {
350  return true;
351  }
352  }
353  return false;
354  }

363  bool GetKeyState(HWND hWnd, vector<DWORD>& UseKeyVec) {
364  m_KeyMessageActive = false;
365  //一つ前にコピー
366  CopyMemory(m_bLastKeyTbl, m_bPushKeyTbl, sizeof(m_bLastKeyTbl));
367  //キーボードの状態を初期化
368  ::ZeroMemory(&m_bPushKeyTbl, sizeof(m_bPushKeyTbl));
369  ::ZeroMemory(&m_bPressedKeyTbl, sizeof(m_bPressedKeyTbl));
370  ::ZeroMemory(&m_bUpKeyTbl, sizeof(m_bUpKeyTbl));
371  //マウスのチェック
372  bool MouseEnabled = IsMouseEnabled(hWnd, UseKeyVec);
373  //自分自身にフォーカスがない場合はfalse
374  if (::GetFocus() != hWnd) {
375  return false;
376  }
377  SHORT ret = GetAsyncKeyState((int)VK_ESCAPE);
378  if (ret & 0x8000) {
379  //Escキーは無条件にfalse
380  return false;
381  }
382  size_t sz = UseKeyVec.size();
383  for (size_t i = 0; i < sz; i++) {
384  ret = GetAsyncKeyState((int)UseKeyVec[i]);
385  if (UseKeyVec[i] == VK_LBUTTON || UseKeyVec[i] == VK_RBUTTON || UseKeyVec[i] == VK_MBUTTON) {
386  if (!MouseEnabled) {
387  //マウスが無効ならUsedに入っていても無視
388  continue;
389  }
390  }
391  if (ret & 0x8000) {
392  m_bPushKeyTbl[UseKeyVec[i]] = true;
393  m_KeyMessageActive = true; //メッセージをゲームが受け取る
394  if (!m_bLastKeyTbl[UseKeyVec[i]]) {
395  //前回押されてなくて今回押された
396  m_bPressedKeyTbl[UseKeyVec[i]] = true;
397  }
398  }
399  else {
400  //キーは押されてない
401  if (m_bLastKeyTbl[UseKeyVec[i]]) {
402  //前回押されていて今回押されてない
403  m_bUpKeyTbl[UseKeyVec[i]] = true;
404  m_KeyMessageActive = true; //メッセージをゲームが受け取る
405  }
406  }
407  }
408  return m_KeyMessageActive;
409  }

456  void ResetControlerState() {
457  DWORD dwResult;
458  for (DWORD i = 0; i < MAX_CONTROLLERS; i++)
459  {
460  //一つ前の状態を保存
461  m_State[i].wLastButtons = m_State[i].Gamepad.wButtons;
462  // XInputから現在のステータスを得る
463  XINPUT_STATE State;
464  dwResult = XInputGetState(i, &State);
465  if (dwResult == ERROR_SUCCESS) {
466  m_State[i].dwPacketNumber = State.dwPacketNumber;
467  m_State[i].Gamepad = State.Gamepad;
468  m_State[i].bConnected = true;
469  //ボタンの設定
470  //コピー
471  m_State[i].wButtons = m_State[i].Gamepad.wButtons;
472  //更新ボタン
473  m_State[i].wNowUpdateButtons = m_State[i].Gamepad.wButtons;
474  m_State[i].bLeftTrigger = m_State[i].Gamepad.bLeftTrigger;
475  m_State[i].bRightTrigger = m_State[i].Gamepad.bRightTrigger;
476  //前回より変化しているボタンのみフラグを立てる
477  //XOR演算により実装
478  m_State[i].wNowUpdateButtons ^= m_State[i].wLastButtons;
479  //押された瞬間をセット
480  m_State[i].wPressedButtons = m_State[i].wNowUpdateButtons & m_State[i].wButtons;
481  //離された瞬間をセット
482  m_State[i].wReleasedButtons = m_State[i].wNowUpdateButtons & (m_State[i].wButtons ^ 0xffff);
483  //レバーのデッドゾーンの設定
484  if ((m_State[i].Gamepad.sThumbLX < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE &&
485  m_State[i].Gamepad.sThumbLX > -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE) &&
486  (m_State[i].Gamepad.sThumbLY < XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE &&
487  m_State[i].Gamepad.sThumbLY > -XINPUT_GAMEPAD_LEFT_THUMB_DEADZONE))
488  {
489  m_State[i].Gamepad.sThumbLX = 0;
490  m_State[i].Gamepad.sThumbLY = 0;
491  }
492  if (m_State[i].Gamepad.sThumbLX >= 0) {
493  m_State[i].fThumbLX = (float)m_State[i].Gamepad.sThumbLX / (float)SHRT_MAX;
494  }
495  else {
496  m_State[i].fThumbLX = (float)m_State[i].Gamepad.sThumbLX / (float)(SHRT_MAX + 1);
497  }
498  if (m_State[i].Gamepad.sThumbLY >= 0) {
499  m_State[i].fThumbLY = (float)m_State[i].Gamepad.sThumbLY / (float)SHRT_MAX;
500  }
501  else {
502  m_State[i].fThumbLY = (float)m_State[i].Gamepad.sThumbLY / (float)(SHRT_MAX + 1);
503  }
504 
505  if ((m_State[i].Gamepad.sThumbRX < XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE &&
506  m_State[i].Gamepad.sThumbRX > -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE) &&
507  (m_State[i].Gamepad.sThumbRY < XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE &&
508  m_State[i].Gamepad.sThumbRY > -XINPUT_GAMEPAD_RIGHT_THUMB_DEADZONE))
509  {
510  m_State[i].Gamepad.sThumbRX = 0;
511  m_State[i].Gamepad.sThumbRY = 0;
512  }
513  if (m_State[i].Gamepad.sThumbRX >= 0) {
514  m_State[i].fThumbRX = (float)m_State[i].Gamepad.sThumbRX / (float)SHRT_MAX;
515  }
516  else {
517  m_State[i].fThumbRX = (float)m_State[i].Gamepad.sThumbRX / (float)(SHRT_MAX + 1);
518  }
519  if (m_State[i].Gamepad.sThumbRY >= 0) {
520  m_State[i].fThumbRY = (float)m_State[i].Gamepad.sThumbRY / (float)SHRT_MAX;
521  }
522  else {
523  m_State[i].fThumbRY = (float)m_State[i].Gamepad.sThumbRY / (float)(SHRT_MAX + 1);
524  }
525  }
526  else {
527  m_State[i].bConnected = false;
528  }
529  }
530  }
