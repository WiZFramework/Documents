939  auto Ptr = dynamic_pointer_cast< BackupData<T> >(m_MeshPrimData.m_BackUpData);
940  if (!Ptr) {

944  L"MeshResource::UpdateVirtexBuffer<T>()"
945  );
946  }
947 
948  if (NewBuffer.size() != Ptr->m_Vertices.size()) {
949  // Map失敗
950  throw BaseException(
951  L"変更する頂点の数がバックアップと違います",
952  L"if (NewBuffer.size() != Ptr->m_Vertices.size())",
953  L"MeshResource::UpdateVirtexBuffer<T>()"
954  );
955  }
956  //座標を変更する
957  auto Dev = App::GetApp()->GetDeviceResources();
958  auto pID3D11DeviceContext = Dev->GetD3DDeviceContext();
959  //頂点バッファをリソースから取り出す
960  auto pVertexBuffer = GetVertexBuffer().Get();
961 
962  //D3D11_MAP_WRITE_DISCARDは重要。この処理により、GPUに邪魔されない
963  D3D11_MAP mapType = D3D11_MAP_WRITE_DISCARD;
964  D3D11_MAPPED_SUBRESOURCE mappedBuffer;
965  //頂点のマップ
966  if (FAILED(pID3D11DeviceContext->Map(pVertexBuffer, 0, mapType, 0, &mappedBuffer))) {
967  // Map失敗
968  throw BaseException(
969  L"頂点のMapに失敗しました。",
970  L"if(FAILED(pID3D11DeviceContext->Map()))",
971  L"MeshResource::UpdateVirtexBuffer<T>()"
972  );
973  }
974  //頂点の変更
975  T* vertices = (T*)mappedBuffer.pData;
976  for (size_t i = 0; i < NewBuffer.size(); i++) {
977  vertices[i] = NewBuffer[i];
978  }
979  //アンマップ
980  pID3D11DeviceContext->Unmap(pVertexBuffer, 0);
981 
982  }
