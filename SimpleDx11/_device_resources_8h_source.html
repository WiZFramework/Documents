991  void GetLocalPositions(vector<bsm::Vec3>& ret)const {
992  //ワールド行列
993  bsm::Mat4x4 world;
994  if (m_MeshPrimData.m_UseMeshToTransformMatrix) {
995  //メッシュが変換行列を持っている場合
996  world = GetMeshToTransformMatrix();
997  }
998  else {
999  world.identity();
1000  }
1001  ret.clear();
1002  for (auto& v : m_MeshPrimData.m_Vertices) {
1003  bsm::Vec3 pos;
